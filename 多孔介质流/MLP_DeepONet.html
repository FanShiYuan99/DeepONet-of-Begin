class MLP_Block(nn.Module):
    def __init__(self, input_dim, hidden_channel, output_dim, hidden_size=2):
        super(MLP_Block, self).__init__()
        self.activation = nn.Tanh()
        self.IN = nn.Linear(input_dim, hidden_channel)
        self.hidden_layers = nn.ModuleList([
            nn.Linear(hidden_channel, hidden_channel) for _ in range(hidden_size)
        ])
        self.OUT = nn.Linear(hidden_channel, output_dim)
        self._init_weights()

    def _init_weights(self):
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.normal_(m.weight, mean=0.0, std=0.1)
                nn.init.constant_(m.bias, 0.0)

    def forward(self, x):
        x = self.activation(self.IN(x))
        for layer in self.hidden_layers:
            x = self.activation(layer(x))
        x = self.OUT(x)
        return x

class Modified_MLP_Block(nn.Module):
    def __init__(self, input_dim, hidden_channel, output_dim, hidden_size=4):
        super(Modified_MLP_Block, self).__init__()
        self.activation = nn.Tanh()
        self.encodeU = nn.Linear(input_dim, hidden_channel)
        self.encodeV = nn.Linear(input_dim, hidden_channel)
        self.In = nn.Linear(input_dim, hidden_channel)

        self.hidden_layers = nn.ModuleList([
            nn.Linear(hidden_channel, hidden_channel) for _ in range(hidden_size)
        ])
        self.out = nn.Linear(hidden_channel, output_dim)
        self._init_weights()

    def _init_weights(self):
        for m in self.modules():
            if isinstance(m, nn.Linear):
                nn.init.normal_(m.weight, mean=0.0, std=0.1)
                nn.init.constant_(m.bias, 0.0)

    def forward(self, x):
        U = self.activation(self.encodeU(x))
        V = self.activation(self.encodeV(x))
        Hidden = self.activation(self.In(x))

        for layer in self.hidden_layers:
            Z = self.activation(layer(Hidden))
            Hidden = (1 - Z) * U + Z * V

        x = self.out(Hidden)
        return x

class DeepONet(nn.Module):
    def __init__(self, branch_input_dim, trunk_input_dim, hidden_channel, dim_p=1):
        super(DeepONet, self).__init__()
        self.dim_p = dim_p

        self.branch_net = Modified_MLP_Block(branch_input_dim, hidden_channel, 40, 4)
        self.trunk_net  = Modified_MLP_Block(trunk_input_dim, hidden_channel, 40, 4)

    def forward(self, branch_input, trunk_input):
        B_f = self.branch_net(branch_input)
        T_f = self.trunk_net(trunk_input)
        u_out = torch.sum(B_f * T_f, dim=-1, keepdim=True)
        return u_out

class SDeepONet(nn.Module):

class get_Dataset(Dataset):
    def __init__(self, data, norm_):
        self.data = data
        self.norm_stats = norm_

    def __getitem__(self, index):
        idx_data = self.data[index]
        x = (idx_data[0] - self.norm_stats['x_mean']) / self.norm_stats['x_std']
        y = (idx_data[1] - self.norm_stats['y_mean']) / self.norm_stats['y_std']
        p = (idx_data[2] - self.norm_stats['p_mean']) / self.norm_stats['p_std']
        feature = idx_data[3]

        B_data = torch.tensor(feature, dtype=torch.float32, requires_grad=True)
        T_data = torch.tensor([x,y], dtype=torch.float32, requires_grad=True)
        P_data = torch.tensor([p], dtype=torch.float32, requires_grad=True)
        return B_data, T_data, P_data

    def __len__(self):
        return len(self.data)

class ConvAE(nn.Module):
    def __init__(self, latent_dim=16):
        super(ConvAE, self).__init__()

        # 100100  1313
        self.encoder = nn.Sequential(
            nn.Conv2d(1, 8, kernel_size=3, stride=2, padding=1),   # (8, 50, 50)
            nn.ReLU(),
            nn.Conv2d(8, 16, kernel_size=3, stride=2, padding=1),  # (16, 25, 25)
            nn.ReLU(),
            nn.Conv2d(16, 32, kernel_size=3, stride=2, padding=1), # (32, 13, 13)
            nn.ReLU(),
        )
        self.flatten = nn.Flatten()
        self.fc_enc = nn.Linear(32 * 13 * 13, latent_dim)

        # 1313  100100
        self.fc_dec = nn.Linear(latent_dim, 32 * 13 * 13)
        self.decoder = nn.Sequential(
            nn.ConvTranspose2d(32, 16, kernel_size=4, stride=2, padding=1),  # (16, 26, 26)
            nn.ReLU(),
            nn.ConvTranspose2d(16, 8, kernel_size=4, stride=2, padding=1),   # (8, 52, 52)
            nn.ReLU(),
            nn.ConvTranspose2d(8, 1, kernel_size=4, stride=2, padding=1),    # (1, 104, 104)
            nn.Sigmoid()
        )

    def forward(self, x):
        batch_size = x.shape[0]
        h = self.encoder(x)
        z = self.fc_enc(self.flatten(h))

        x_hat = self.fc_dec(z).view(batch_size, 32, 13, 13)
        x_hat = self.decoder(x_hat)

        x_hat = x_hat[:, :, :100, :100]
        return x_hat, z

def train_ae(train_model, data_loader, device='cuda', train_epochs=30, lr=1e-3,
             save_path=r"F:\Pycharm\Pycharm_Files\Pytorch\DeepLearning\Random_Param\trained_ae.pth"):
    train_model = train_model.to(device)
    optimizer = torch.optim.Adam(train_model.parameters(), lr=lr)
    loss_fn = nn.MSELoss()

    for epoch in range(train_epochs):
        train_model.train()
        total_loss = 0
        for batch in tqdm(data_loader, desc=f"Epoch {epoch+1}/{epochs}"):
            x = batch[0].to(device)
            optimizer.zero_grad()
            x_rec, _ = train_model(x)
            loss = loss_fn(x_rec, x)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        print(f"Epoch {epoch+1}, Loss: {total_loss/len(data_loader):.6f}")

    torch.save(train_model.state_dict(), save_path)

def load_ae_data(folder, max_samples=1000):
    paths = sorted(Path(folder).glob("k_field_*.csv"))[:max_samples]
    data = []
    for path in tqdm(paths, desc="Loading k_field"):
        k = pd.read_csv(path)['k'].values.reshape(100, 100) / 12.0  # 0~1
        data.append(k)
    data = np.array(data, dtype=np.float32)
    data = data[:, np.newaxis, :, :]  # shape = (N, 1, 100, 100)
    return torch.tensor(data)

def get_ae_feature(k_array, is_load=True, is_train=False, latent_dim=100,
             save_path="F:/Pycharm/Pycharm_Files/Pytorch/DeepLearning/Random_Param/trained_ae.pth"):
    """
     k_array: shape  (100, 100)  (N, 100, 100)
     latent : shape  (latent_dim,)  (N, latent_dim)
    """
    if is_train:
        k_tensor = load_ae_data("F:/DeepLearningDataset/Random_Flow/k_filed", max_samples=1100)
        dataloader = DataLoader(TensorDataset(k_tensor), batch_size=128, shuffle=True)

        train_model = ConvAE(latent_dim=latent_dim)
        train_ae(train_model, dataloader, train_epochs=100, lr=1e-3, save_path=save_path)

    if not hasattr(get_ae_feature, "ae_model"):
        get_ae_feature.ae_model = ConvAE(latent_dim=latent_dim)
        if is_load:
            get_ae_feature.ae_model.load_state_dict(torch.load(save_path))
        get_ae_feature.ae_model.eval()

    if isinstance(k_array, np.ndarray):
        if k_array.ndim == 2:
            k_array = k_array[np.newaxis, np.newaxis, :, :]  # (1, 1, 100, 100)
        elif k_array.ndim == 3:
            k_array = k_array[:, np.newaxis, :, :]           # (N, 1, 100, 100)
        else:
            raise ValueError(f" shape (100,100)(N,100,100){k_array.shape}")
    else:
        raise TypeError("k_array  numpy.ndarray")

    with torch.no_grad():
        tensor = torch.tensor(k_array / 12.0, dtype=torch.float32)
        _, latent = get_ae_feature.ae_model(tensor)
        return latent.numpy() if latent.shape[0] > 1 else latent.squeeze(0).numpy()

class PINN_Vibrate:
    def __init__(self, train_model, batch_size = 10000, learning_rate = 0.001, step = 200, gamma = 0.95, latent_dim=100):
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        print(self.device)
        self.model = train_model.to(self.device)
        self.batch_size = batch_size
        self.criterion = nn.MSELoss()
        self.optimizer = torch.optim.Adam(self.model.parameters(), lr=learning_rate)
        self.scheduler = torch.optim.lr_scheduler.StepLR(self.optimizer, step_size=step, gamma=gamma)
        self.Losses = []
        self.Lambda = []
        self.file_name = r"F:\Pycharm\Pycharm_Files\Pytorch\DeepLearning\Random_Param\Flow.pkl"
        self.train_set, self.test_set, self.Train_DataLoader, self.Test_DataLoader, self.norm_stats = self.load_train_set(latent_dim=latent_dim)

    def load_model(self):
        self.model.load_state_dict(torch.load(self.file_name, weights_only=True))
        print('model loaded')
        
    def u_function(self, B_data, T_data):
        p = self.model(B_data, T_data)
        return p
        
    def get_data_loss(self, B_data, T_data, P_data):
        p_pre = self.u_function(B_data, T_data)
        p_ex = P_data[:,0:1]
        loss_p = self.criterion(p_pre, p_ex)
        return loss_p

    def train(self, train_epochs):
        self.Losses.append(['total_loss'])
        print('training started')
        start_time = datetime.datetime.now()

        for epoch in range(train_epochs):
            item_loss = 0.
            for i, (B_data, T_data, P_data) in enumerate(self.Train_DataLoader):
                B_data, T_data, P_data = B_data.to(self.device), T_data.to(self.device), P_data.to(self.device)

                self.optimizer.zero_grad()
                loss = self.get_data_loss(B_data, T_data, P_data)
                loss.backward()
                self.optimizer.step()
                item_loss += loss.item()
            self.Losses.append([
                item_loss/len(self.Train_DataLoader)
            ])
            self.scheduler.step()
            if epoch % 10 == 0 or epoch == train_epochs - 1:
                print(f"[Epoch {epoch+1}/{train_epochs}] [loss: {item_loss/len(self.Train_DataLoader):.8f}]")
                torch.save(self.model.state_dict(), self.file_name)
        torch.save(self.model.state_dict(), self.file_name)
        print("Current learning rate:", self.optimizer.param_groups[0]['lr'])
        print("Training Time:", (datetime.datetime.now() - start_time).total_seconds() / 3600, "h")

    def plot_loss(self):
        if not self.Losses:
            return
        plt.title('log(loss)')
        temp = np.array(self.Losses[1:])
        for i in range(temp.shape[1]):
            plt.plot(temp[:, i].flatten(), label=self.Losses[0][i])
        plt.yscale('log')
        plt.legend()
        plt.show()

    def predict(self, B_data, T_data):
        p_pre = self.u_function(B_data, T_data)
        p_pre = p_pre.cpu().detach().numpy()
        return p_pre

    def compute_samplewise_error(self, dataloader, u_function, criterion):
        p_errors = []
        for B_data, T_data, P_data in dataloader:
            B_data, T_data, P_data = B_data.to(self.device), T_data.to(self.device), P_data.to(self.device)
            p_ex = P_data[:, 0:1]
            p_pre = u_function(B_data, T_data)
            p_loss = criterion(p_pre, p_ex)
            p_errors.append(p_loss.detach().cpu().item())
        return p_errors

    def get_evaluate(self):
        print("Evaluating on Train Set...")
        train_errors = self.compute_samplewise_error(self.Train_DataLoader, self.u_function, self.criterion)
        print(f'Train Set - Batch Mean Pressure Loss: {sum(train_errors)/len(train_errors):.6f}')

        print("Evaluating on Test Set...")
        test_errors = self.compute_samplewise_error(self.Test_DataLoader, self.u_function, self.criterion)
        print(f'Test Set - Batch Mean Pressure Loss: {sum(test_errors)/len(test_errors):.6f}')

        # 
        plt.figure(figsize=(6, 5))

        # Pressure loss 
        ax1 = plt.subplot(1, 1, 1)
        ax2 = ax1.twinx()

        ax1.hist(train_errors, bins=30, alpha=0.6, label='Train', color='steelblue')
        ax2.hist(test_errors, bins=30, alpha=0.6, label='Test', color='orange')

        ax1.set_xlabel("Loss")
        ax1.set_ylabel("Train Frequency", color='steelblue')
        ax2.set_ylabel("Test Frequency", color='orange')
        ax1.set_title("Frequency Loss Distribution")

        # 
        lines, labels = ax1.get_legend_handles_labels()
        lines2, labels2 = ax2.get_legend_handles_labels()
        ax1.legend(lines + lines2, labels + labels2, loc='upper right')

        plt.tight_layout()
        plt.show()
        return train_errors, test_errors

    def get_predict(self, B_data, T_data):
        p_pre = self.u_function(B_data, T_data)
        p_pre = p_pre * self.norm_stats['p_std'] + self.norm_stats['p_mean']
        return p_pre.cpu().detach().numpy().reshape(-1,1)

    def get_exact(self, P_data):
        p_ex = P_data[:,0:1]
        p_ex = p_ex * self.norm_stats['p_std'] + self.norm_stats['p_mean']
        return p_ex.cpu().detach().numpy().reshape(-1,1)

    def plot_result(self, num=0):
        B_data, T_data, P_data, params_df = self.get_test_data(num)
        plt.figure(figsize=(6,5))
        plt.imshow(params_df, cmap='viridis')
        plt.colorbar()
        plt.title("Random permeability b(x,y)")
        plt.show()

        B_data, T_data, P_data = B_data.to(self.device), T_data.to(self.device), P_data.to(self.device)

        p_pre = self.get_predict(B_data, T_data).flatten()
        p_ex = self.get_exact(P_data).flatten()
        x_np = (T_data[:,0:1].cpu().detach().numpy().reshape(-1,1) * self.norm_stats['x_std'] + self.norm_stats['x_mean']).flatten()
        y_np = (T_data[:,1:2].cpu().detach().numpy().reshape(-1,1) * self.norm_stats['y_std'] + self.norm_stats['y_mean']).flatten()

        p_err = np.abs(p_ex - p_pre)

        print(f'RMSE P = {np.linalg.norm(p_err)/np.linalg.norm(p_ex)} MAE P = {np.mean(np.abs(p_err))}')
        plt.figure(figsize=(18,5))
        for i, temp in enumerate([[x_np,y_np,p_pre,'P Pre'],[x_np,y_np,p_ex,'P Exact'],[x_np,y_np,p_err,'P Error']]):
            plt.subplot(1,3,i+1)
            tri = plt.tricontourf(temp[0], temp[1], temp[2], levels=100, cmap='bwr')
            plt.colorbar(tri)
            plt.xlabel('x')
            plt.ylabel('y')
            plt.title(temp[3])
            plt.tight_layout()
        plt.show()
        
    def get_test_data(self, number=0):
        sampled_data, name = self.test_set[number]
        params_folder = Path(r"F:\DeepLearningDataset\Random_Flow\k_filed")
        params_df = pd.read_csv(params_folder / name).dropna()['k'].values.reshape(100,100)
        x = (sampled_data[['x']] - self.norm_stats['x_mean']) / self.norm_stats['x_std']
        y = (sampled_data[['y']] - self.norm_stats['y_mean']) / self.norm_stats['y_std']
        p = (sampled_data[['p']] - self.norm_stats['p_mean']) / self.norm_stats['p_std']
        B_data_single = sampled_data['features'].values[0]
        B_data = np.stack([B_data_single] * len(sampled_data), axis=0)
        T_data = np.hstack((x.values.reshape(-1,1),y.values.reshape(-1,1)))
        P_data = p.values.reshape(-1,1)
        B_data = torch.tensor(B_data, dtype=torch.float32, requires_grad=True).to(self.device)
        T_data = torch.tensor(T_data, dtype=torch.float32, requires_grad=True).to(self.device)
        P_data = torch.tensor(P_data, dtype=torch.float32, requires_grad=True).to(self.device)
        return B_data, T_data, P_data, params_df
        
    def load_train_set(self, latent_dim=100):
        data_set = []
        params_folder = Path(r"F:\DeepLearningDataset\Random_Flow\k_filed")
        results_folder = Path(r"F:\DeepLearningDataset\Random_Flow\results")

        params_list = sorted(list(params_folder.glob('k_field_*.csv')))
        results_list = sorted(list(results_folder.glob('result_*.csv')))

        for params_path, results_path in zip(params_list, results_list):
            params_df = pd.read_csv(params_path).dropna()['k'].values.reshape(100,100)
            results_df = pd.read_csv(results_path, header=None).dropna().rename(columns={0:'x',1:'y',2:'p'})

            params_feature = get_ae_feature(params_df, is_load=True, is_train=False, latent_dim=latent_dim)
            results_df['features'] = [params_feature for _ in range(len(results_df))]
            data_set.append((results_df, params_path.name))
        l = len(data_set)
        print(l)
        random.seed(42)
        random.shuffle(data_set)
        train_set = data_set[:int(l * 0.8)]
        test_set = data_set[int(l * 0.8):]

        all_train_df = pd.concat([d[0] for d in train_set], ignore_index=True)
        norm_stats = {
            'x_mean': all_train_df['x'].mean(), 'x_std':  all_train_df['x'].std(),
            'y_mean': all_train_df['y'].mean(), 'y_std':  all_train_df['y'].std(),
            'p_mean': all_train_df['p'].mean(), 'p_std':  all_train_df['p'].std(),
        }

        Train_Dataset = get_Dataset(np.array(pd.concat([d[0] for d in train_set], ignore_index=True)), norm_stats)
        Test_Dataset = get_Dataset(np.array(pd.concat([d[0] for d in test_set], ignore_index=True)), norm_stats)
        Train_DataLoader = DataLoader(Train_Dataset, batch_size=self.batch_size, shuffle=True)
        Test_DataLoader = DataLoader(Test_Dataset, batch_size=self.batch_size, shuffle=False)
        print(f'Train Length: {len(Train_DataLoader)}, Test Length: {len(Test_DataLoader)}')

        return train_set, test_set, Train_DataLoader, Test_DataLoader, norm_stats
